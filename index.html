/** * ========================================================================= * SIPP — ENGINE DE PILOTAGE DES VENTES MINT & CONSO PILOTE (V2.0) * ========================================================================= * Système unifié : Traitement multi-feuilles, Rapprochement par ID contrat, * Règle de non-réversibilité comptable & Moteur transactionnel sécurisé. */ // Initialisation sécurisée de l'état global de l'application if (!window.appContractsState) { window.appContractsState = {}; } /** * 1. CLASSIFICATION INTELLIGENTE DU STATUT "PAYÉ" * Gère les variantes d'écriture, accents, et casses. */ function isStatusPaid(statusValue) { if (statusValue === undefined || statusValue === null) return false; const normalized = String(statusValue) .trim() .toLowerCase() .normalize("NFD") .replace(/[\u0300-\u036f]/g, ""); // Supprime les accents (ex: "payé" -> "paye") const paidKeywords = ["paye", "payee", "regle", "solde", "encaisse", "valide", "complete"]; return paidKeywords.some(keyword => normalized.includes(keyword)) && !normalized.includes("non"); } /** * 2. CLASSIFICATION DU STATUT "ANNULÉ" */ function isStatusCancelled(statusValue) { if (statusValue === undefined || statusValue === null) return false; const normalized = String(statusValue).trim().toLowerCase(); return normalized.includes("annule") || normalized.includes("resilie") || normalized.includes("rejete"); } /** * 3. EXTRACTION ROBUSTE DES NUMÉROS DE CONTRAT (5 à 7 chiffres) */ function extractContractId(value) { if (!value) return null; const cleaned = String(value).trim(); const match = cleaned.match(/\b\d{5,7}\b/); return match ? match[0] : null; } /** * 4. MOTEUR TRANSACTIONNEL ET RAPPROCHEMENT DES DONNÉES * Traite les lignes de toutes les feuilles et applique les règles de gestion. */ function processTransactionalUpdate(allRows, currentContractsState) { // Étape d'intégrité : Copie profonde de l'état pour éviter les écritures partielles en cas de plantage const stateClone = JSON.parse(JSON.stringify(currentContractsState)); const report = { success: false, processedCount: 0, alertsRaised: 0, errors: [], warnings: [] }; try { // Étape A : Cartographie et indexation des lignes par numéro de contrat unique allRows.forEach((row, index) => { // Mapping ultra-flexible des en-têtes (tolérance aux libellés modifiés ou tronqués) const rawContractField = row['Numéro Contrat Energie'] || row['Contrat ID'] || row['REF SOUSCRIPTION'] || row['REF SOUSCRIPTI'] || row['Numero Contrat']; const contractId = extractContractId(rawContractField); if (!contractId) { // Ligne ignorée car pas d'ID de contrat valide détecté return; } const rawStatus = row['Statut Vente'] || row['Statut'] || row['status'] || row['Etat']; const paymentValue = parseFloat(row['Montant Brut'] || row['Paiement Brut'] || row['Montant'] || row['Comissions'] || 0); const agentName = row['Agent'] || row['Nom Agent'] || "Inconnu"; // Étape B : Initialisation du contrat dans le dictionnaire s'il n'existe pas encore if (!stateClone[contractId]) { stateClone[contractId] = { id: contractId, agent: agentName, paiementBrut: 0, statutInterne: "EN_ATTENTE", revueQualiteRequise: false, historiqueStatuts: [] }; } const currentContract = stateClone[contractId]; // Étape C : Application de la règle de NON-RÉVERSIBILITÉ const alreadyPaid = currentContract.statutInterne === "PAYE_CONFIRME" || currentContract.paiementBrut > 0; if (isStatusPaid(rawStatus)) { if (isNaN(paymentValue) || paymentValue <= 0) { report.warnings.push(`Ligne ${index + 2} : Contrat ${contractId} détecté "Payé" mais montant brut invalide ou nul.`); } else { // Si déjà payé, on évite d'écraser inutilement sauf si le montant change currentContract.paiementBrut = paymentValue; currentContract.statutInterne = "PAYE_CONFIRME"; currentContract.revueQualiteRequise = false; // Plus d'alerte si re-confirmé payé report.processedCount++; } } else if (isStatusCancelled(rawStatus)) { if (alreadyPaid) { // RÈGLE MÉTIER CRITIQUE : Un contrat déjà payé qui passe en "Annulé" lève une alerte Revue Qualité au lieu d'être effacé currentContract.revueQualiteRequise = true; currentContract.statutInterne = "ALERTE_ANNULATION_POST_PAIEMENT"; if (!currentContract.historiqueStatuts.includes("ALERTE_ANNULATION")) { currentContract.historiqueStatuts.push("ALERTE_ANNULATION"); } report.alertsRaised++; } else { // Si non payé préalablement, l'annulation automatique suit son cours standard currentContract.paiementBrut = 0; currentContract.statutInterne = "ANNULE_CRM"; } } else { // Autres statuts (en cours, etc.) : On ne touche pas au montant payé s'il existe déjà if (!alreadyPaid) { currentContract.statutInterne = "EN_COURS_CRM"; } } }); report.success = true; return { updatedState: stateClone, report: report }; } catch (globalError) { report.success = false; report.errors.push({ global: true, error: globalError.message }); return { updatedState: currentContractsState, report: report }; } } /** * 5. VALIDATION POST-TRAITEMENT ET COMPARAISON DE COHÉRENCE */ function validatePostProcessing(allRows, updatedState) { const auditReport = { expectedPayments: 0, confirmedPayments: 0, discrepancies: [] }; allRows.forEach((row, index) => { const rawContractField = row['Numéro Contrat Energie'] || row['Contrat ID'] || row['REF SOUSCRIPTION'] || row['REF SOUSCRIPTI']; const contractId = extractContractId(rawContractField); const rawStatus = row['Statut Vente'] || row['Statut'] || row['status']; if (contractId && isStatusPaid(rawStatus)) { auditReport.expectedPayments++; const stateRecord = updatedState[contractId]; if (stateRecord && stateRecord.paiementBrut > 0) { auditReport.confirmedPayments++; } else { auditReport.discrepancies.push({ line: index + 2, contractId: contractId, reason: `Détecté "Payé" dans l'import, mais le montant brut du contrat n'est pas actif dans le système.` }); } } }); return auditReport; } /** * 6. INITIALISATION DU GESTIONNAIRE D'IMPORT (MULTI-FEUILLES EXCEL) */ document.addEventListener("DOMContentLoaded", () => { const fileInput = document.getElementById('file-input') || document.querySelector('input[type="file"]'); const errorDisplay = document.querySelector('.error-message') || document.getElementById('error-display') || document.querySelector('.bg-red-900\\/50') || document.querySelector('[class*="red"]'); if (!fileInput) { console.error("SIPP Engine : Aucun sélecteur de fichier d'import trouvé dans le DOM."); return; } fileInput.addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(evt) { try { if (typeof XLSX === 'undefined') { throw new Error("Bibliothèque d'analyse Excel absente. Assurez-vous d'avoir importé le script CDN SheetJS."); } const data = evt.target.result; // Lecture globale de tous les formats Excel (xls, xlsx, xlsm, xlsb) const workbook = XLSX.read(data, { type: 'binary' }); let combinedRows = []; // Étape 1 : Lecture exhaustive de CHAQUE feuille de calcul workbook.SheetNames.forEach(sheetName => { const worksheet = workbook.Sheets[sheetName]; const sheetData = XLSX.utils.sheet_to_json(worksheet, { defval: null }); // Fusion des données de toutes les feuilles en un seul tableau combinedRows = combinedRows.concat(sheetData); }); console.log(`[SIPP Audit] ${combinedRows.length} lignes extraites à partir de ${workbook.SheetNames.length} feuilles.`); // Étape 2 : Lancement de la mise à jour transactionnelle sécurisée const result = processTransactionalUpdate(combinedRows, window.appContractsState); if (result.report.success && result.report.errors.length === 0) { // Étape 3 : Validation post-traitement const audit = validatePostProcessing(combinedRows, result.updatedState); if (audit.discrepancies.length === 0) { // VALIDATION validée : On applique définitivement les modifications window.appContractsState = result.updatedState; // Nettoyage de l'affichage d'erreur s'il existait if (errorDisplay) { errorDisplay.style.display = 'none'; errorDisplay.innerHTML = ''; } // Notification de réussite complète alert(`✅ Importation Excel terminée avec succès !\n\n` + `• Feuilles analysées : ${workbook.SheetNames.length}\n` + `• Total lignes traitées : ${combinedRows.length}\n` + `• Contrats enregistrés/mis à jour : ${result.report.processedCount}\n` + `• Alertes Qualité levées (non-réversibilité) : ${result.report.alertsRaised}`); console.log("Nouvel état applicatif (window.appContractsState) :", window.appContractsState); } else { console.warn("Divergences détectées lors de l'audit post-traitement :", audit.discrepancies); alert(`⚠ Importation partielle avec avertissements.\n` + `Certains statuts payés n'ont pas pu être affectés. Consultez la console de développement pour voir le rapport d'anomalies.`); } } else { // ROLLBACK : Aucune modification n'a été appliquée au state original throw new Error(result.report.errors[0]?.error || "Erreur interne pendant le calcul."); } } catch (error) { console.error("[SIPP Error]", error); if (errorDisplay) { errorDisplay.style.display = 'block'; errorDisplay.innerHTML = `⚠ Impossible de lire le fichier : ${error.message}`; } alert(`❌ Échec de l'importation :\n${error.message}`); } }; reader.readAsBinaryString(file); }); });
